def miniMaxSum(arr):
    arr.sort()

    # Calculate the sum of all elements
    total_sum = sum(arr)

    # Calculate the minimum and maximum sums
    min_sum = total_sum - arr[-1]
    max_sum = total_sum - arr[0]

    # Print the results
    print(min_sum, max_sum)
